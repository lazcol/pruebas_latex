\begin{section}{Ley de Charles Augustin de COULOMB}
\begin{displaymath}
\begin{array}{ccc}
		\text{Campo Eléctrico} & & \text{Fuerza Eléctrica} \\
	\vec{E}=k\frac{q}{d^2} & & \vec{F}=k\frac{q\cdot Q}{d^2}
\end{array}
\end{displaymath}
\end{section}
\begin{section}{Ley de Carl Friedrich GAUSS}
	\begin{displaymath}
	\oiint_{SG} \vec{E} \cdot \vec{ds} = 4\,\pi\,k_e\,Q^*=\frac{Q^*}{\varepsilon_o}
	\end{displaymath}
\end{section}
\begin{section}{Distribución Discreta de Carga}
	\begin{displaymath}
		\vec{E} = \sum\limits_{i=1}^{n} \frac{k\cdot q_i}{\norma{{r_i}-{r_o}}^2}
	\end{displaymath}
\end{section}
\begin{section}{Distribución Continua de Carga}
	\begin{displaymath}
		\vec{E} = \int_0^Q \frac{k}{\norma{{r}-{r_o}}^2}\cdot \vec{dq}
	\end{displaymath}
\end{section}

\begin{section}{Momento Dipolar}
	$$\vec{p} = q\cdot \vec{d}$$
\end{section}
\begin{section}{Momento Sobre un Dipolo}
	$$\vec{\tau}=\vec{p}\x \vec{E}$$
\end{section}
\begin{section}{Campo Debido a un Dipolo ${}_{CAC}$}
	Ambas cargas sobre el eje $x$ ubicadas en $(-d,0)\wedge (d,0)$ y haciendo la suposición que $x>>d$. En función del momento dipolar
	$$ \vec{E} (x;0) \approx \frac{2\cdot k_e \cdot \vec{p}}{x^3} $$
\end{section}
\begin{section}{Anillo de radio $r$ y carga $Q$ ${}_{CAC}$}
	El anillo se encuentra en el plano $xy$ como $x^2+y^2=r^2$. Se plantea el campo eléctrico sobre puntos del eje $z$ que pasa por el centro geométrico del anillo.
	$$ \vec{E}(0,0,z) = \frac{k\cdot Q \cdot z}{\left( r^2+z^2\right) ^\frac{3}{2}}$$
	Si $z>>r$
	$$ \vec{E}(0,0,z) = \frac{k\cdot Q}{z^2}$$
	Por otro lado, si hacemos tender $z\to 0$ para saber el campo en el centro del anillo
	$$ \vec{E}(0,0,0) =\lim_{z\to0} \frac{k\cdot Q \cdot z}{\left( r^2+z^2\right) ^\frac{3}{2}}=0$$
\end{section}
\begin{section}{Disco de radio $r$ y distribución superficial de carga uniforme $\sigma$ ${}_{CAC}$}
	Disco ubicado en el centro de coordenadas, sobre el plano $xy$. El campo es pedido en puntos del eje $z$ que pasa por el centro geométrico del disco.
	$$ \vec{E}(0,0,z) = 2\pi k_e \sigma \left( 1-\frac{z}{\sqrt{r^2+z^2}}\right)$$
	Si $r>>z$
	$$ \vec{E}(0,0,z) = 2\pi k_e \sigma = \frac{\sigma}{2\epsilon_o}$$
	Si $z>>r$
	$$ \vec{E}(0,0,z) = 2\pi k_e \sigma \left( 1-\frac{1}{z}\right)$$
\end{section}
\begin{section}{Plano $\infty$ ${}_{CAC}\;{}_{CFG}$}
		Suponiendo que el plano sea el plano vertical $xz$ de ecuación $y=0$. Se pide el campo a una distancia $y$ del mismo.
		$$\vec{E}(0,y,0)=2\, \pi \, k_e \, \sigma = \frac{\sigma}{2\epsilon_o}$$
\end{section}
\begin{section}{Campo eléctrico entre dos placas $\infty$ con igual densidad superficial de carga, de cargas opuestas, separadas una distancia $d$}
	Se suman ambos campos por principio de superposición.\\
	Campo entre medio de las placas
	$$\vec{E} = \frac{\sigma}{\epsilon_o}$$
	Campo en la zona exterior de las placas
	$$\vec{E} = 0$$
\end{section}
\begin{section}{Esfera maciza de radio $R$}
Con distribución volumétrica de carga lineal respecto al radio $Q^*=Q\frac{r^3}{R^3}$\\
Para $r<R$
$$\vec{E} = \frac{k_e\;Q\;r }{R^3}\versorr$$
Para $r>R$
$$\vec{E} = \frac{k_e \; Q}{r^2}\versorr$$

Con distribución volumétrica de carga respecto al radio $Q^*=Q\frac{r}{R}$ \\
Para $r<R$
$$\vec{E} = \frac{k_e\,Q}{r\,R}\versorr$$
Para $r>R$
$$\vec{E} = \frac{k_e \; Q}{r^2}\versorr$$
\end{section}
\begin{section}{Cascarón esférico de radio $R$}
Con distribución superficial de carga $\sigma$ tal que la carga es $Q\x 4\pi R^2$\\
Para $r<R$
$$\vec{E} = 0$$
Porque no hay carga encerrada adentro. La superficie Gaussiana que se toma no tiene ninguna carga que haga atravesar un flujo a través de su superficie.\\
Para $r>R$
$$\vec{E} = \frac{4\pi k_e\;\sigma R^2 }{r^2}\versorr$$
\end{section}
\begin{section}{Cable finito con distribución lineal de carga $\lambda$ ${}_{CAC}$}
	Campo tomado sobre puntos del eje $y$ perpendicular al cable que pasa por su centro geométrico, suponiendo que el cable es de largo $L$ y está centrado en los ejes de coordenadas desde $(-\frac{L}{2},0)$ hasta $(+\frac{L}{2},0)$
	$$\vec{E}(0,y) = \frac{\lambda \; k \; L}{y \cdot \sqrt{\parent{\frac{L}{2}^2+y^2}}}$$
\end{section}
\begin{section}{Cable infinito con distribución lineal de carga $\lambda$ ${}_{CAC}$}
	Campo tomado sobre puntos del eje $y$ perpendicular al cable, suponiendo que el cable es de largo $L=\infty$.
	$$\vec{E}(0,y) = \lim\limits_{L\to\infty}\frac{\lambda \; k \; L}{y \cdot \sqrt{\parent{\frac{L}{2}^2+y^2}}}=\frac{2\;k\;\lambda}{y}$$
\end{section}
\begin{section}{Cable infinito con distribución lineal de carga $\lambda$ ${}_{CAG}$}
	Campo tomado a una distancia $r$ del cable, suponiendo que el cable es $\infty$, tomando una superficie Gaussiana cilíndrica alrededor del cable de radio $r$.
	$$\vec{E}(0,y) = \frac{2\;k\;\lambda}{r}$$
\end{section}
\begin{section}{Potencial electrostático debido a una carga puntual}
	$$\Delta V_{ab} = -\int_a^b \vec{E}\cdot \vec{dl}=-\frac{k_e\cdot Q}{r} \bigg|_a^b$$
\end{section}
\begin{section}{Potencial electrostático debido a varias cargas puntuales - Distribución Discreta de Cargas}
	$r_i$ es el vector posición de la carga $Q_i$, y $r_o$ es el vector posición del lugar en donde se quiere calcular el potencial.
	$$V_{\text{en A}} = \sum_{i=1}^{n} V_i =\sum_{i=1}^{n} \frac{k_e \; Q_i}{\|\vec{r_i}-\vec{r_o}\|}$$
\end{section}
\begin{section}{Potencial electrostático debido a Distribución Contínua de Cargas}
	$$V(r) = \int_0^R 	\frac{k_e\;Q}{\|\vec{r}\|^2}dr=\int_0^V dV = \int_0^Q \frac{k_e }{r}dq$$
\end{section}
\begin{section}{Potencial generado por un plano infinito horizontal de ecuación $z=0$}
	$$V=\int_z^0 \frac{\sigma}{2\epsilon_o}dz = -\frac{\sigma\;z}{2\;\epsilon_o}$$
\end{section}
\begin{section}{Potencial generado por una esfera conductora cargada}
	Para $r>R$ siendo $R$ el radio de la esfera y $Q$ la carga total de la esfera
	$$V = \frac{k_e \; Q}{r}$$
	Para $r<<R$ el campo eléctrico es nulo, por eso el potencial eléctrico permanece constante
	$$V = \frac{k_e\;Q}{R}$$ 
\end{section}
\begin{section}{Potencial de un anillo de radio $r$ con carga $Q$}
	El anillo de ecuación $x^2+y^2=r^2$ se encuentra en el plano horizontal con ecuación $z=0$, centrado en el origen, y el potencial está calculado sobre puntos del eje $z$ prpendicular al mismo y que pasa por su centro geométrico.
	$$ V = \frac{k_e \; Q}{\sqrt{r^2+z^2}}$$
\end{section}
\begin{section}{Potencial de un disco de radio $r$ y carga $Q$}
	$$ V = 2k_e\pi \sigma \parent{\sqrt{r^2+z^2}-z}$$
\end{section}
\begin{section}{Campo Eléctrico, Potencial y Energía Potencial Electrostática}
	\begin{eqnarray*}
	\Delta U = W = \int_a^b \vec{F}\cdot \vec{dl} & \Rightarrow & \vec{F} = -\nabla \cdot U\\
	\Delta V_{ab} = -\int_a^b \vec{E}\cdot \vec{dl} & \Rightarrow & \vec{E} = \nabla \cdot V\\
	\vec{F} = q\, \vec{E} \\
	\int V\cdot dq = U & \Rightarrow & dU = V\cdot dq = -\vec{F}\cdot \vec{dl}
	\end{eqnarray*}
\end{section}
